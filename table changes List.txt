CREATE TABLE rosters (
    roster_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    month_year VARCHAR(10) NOT NULL, -- APR2026

    total_days INT NOT NULL,
    working_days INT DEFAULT 0,
    weekoff_days INT DEFAULT 0,
    holiday_days INT DEFAULT 0,

    target_leaves INT DEFAULT 0,
    base_target DECIMAL(10,2) DEFAULT 0.00,
    extra_assigned_hours DECIMAL(10,2) DEFAULT 0.00,
    final_target DECIMAL(10,2) DEFAULT 0.00,

    status ENUM('draft','active','locked') DEFAULT 'draft',

    created_by INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY unique_user_month (user_id, month_year),
    FOREIGN KEY (user_id) REFERENCES tfs_user(user_id)
);

CREATE TABLE roster_days (
    roster_day_id INT AUTO_INCREMENT PRIMARY KEY,
    roster_id INT NOT NULL,
    user_id INT NOT NULL,
    date DATE NOT NULL,

    day_type ENUM('working','weekoff','holiday','leave') DEFAULT 'working',
    shift ENUM('day','night') DEFAULT 'day',

    is_leave TINYINT(1) DEFAULT 0,
    leave_type_id INT NULL,

    is_target_leave TINYINT(1) DEFAULT 0,
    is_planned_leave TINYINT(1) DEFAULT 0,

    is_emergency TINYINT(1) DEFAULT 0,

    remarks VARCHAR(255),

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY unique_user_date (user_id, date),

    FOREIGN KEY (roster_id) REFERENCES rosters(roster_id),
    FOREIGN KEY (user_id) REFERENCES tfs_user(user_id)
);

CREATE TABLE leave_types (
    leave_type_id INT AUTO_INCREMENT PRIMARY KEY,
    leave_type_name VARCHAR(100) NOT NULL,
    leave_code VARCHAR(20) UNIQUE,

    affects_target TINYINT(1) DEFAULT 1,
    requires_approval TINYINT(1) DEFAULT 1,

    max_days_per_month INT DEFAULT 0,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE leaves (
    leave_id INT AUTO_INCREMENT PRIMARY KEY,

    user_id INT NOT NULL,
    leave_type_id INT NOT NULL,

    from_date DATE NOT NULL,
    to_date DATE NOT NULL,
    total_days INT NOT NULL,

    is_half_day TINYINT(1) DEFAULT 0,
    half_day_type ENUM('first','second') DEFAULT NULL,

    reason TEXT,

    is_planned_leave TINYINT(1) DEFAULT 0,
    is_emergency TINYINT(1) DEFAULT 0,

    is_target_leave TINYINT(1) DEFAULT 0,

    added_date DATE NOT NULL,

    status ENUM('pending','approved','rejected','cancelled') DEFAULT 'pending',

    applied_by INT,
    approved_by INT,
    approved_at TIMESTAMP,
    approval_remarks VARCHAR(255),

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES tfs_user(user_id),
    FOREIGN KEY (leave_type_id) REFERENCES leave_types(leave_type_id)
);
